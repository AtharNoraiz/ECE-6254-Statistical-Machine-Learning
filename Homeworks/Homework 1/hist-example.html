import numpy as np
import matplotlib.pyplot as plt

def histo(n):
    Z = [0] * 1000
    for i in range(1000):
        X = np.random.randn(n)
        Z[i] = max(X)
    plt.hist(Z)
    plt.title("Histogram of Z")
## Show the plot
    plt.show()
